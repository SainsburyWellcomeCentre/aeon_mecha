66def print_current_jobs(): 

67 """ 

68 Return a pandas.DataFrame on the status of each table currently being processed 

69 

70 table | reserve_count | error_count | oldest_job | newest_job 

71 - table: {schema}.{table} name 

72 - reserve_count: number of workers currently working on this table 

73 - error_count: number of jobs errors for this table 

74 - oldest_job: timestamp of the oldest job currently being worked on 

75 - newest_job: timestamp of the most recent job currently being worked on 

76 

77 Provide insights into the current status of the workers 

78 

79 One caveat in this function is that we don't know how many workers are being deployed, 

80 and how they're orchestrated. We can infer by taking the sum of the reserved jobs, 

81 but this won't reflect idling workers because there's no "key_source" to work on for some 

82 particular tables 

83 """ 

84 job_status = [] 

85 for pipeline_module in (lab, subject, experiment, tracking, analysis): 

86 reserved = dj.U('table_name').aggr(pipeline_module.schema.jobs & 'status = "reserved"', 

87 reserve_count='count(table_name)', 

88 oldest_job='MIN(timestamp)', 

89 newest_job='MAX(timestamp)') 

90 errored = dj.U('table_name').aggr(pipeline_module.schema.jobs & 'status = "error"', 

91 error_count='count(table_name)') 

92 jobs_summary = reserved.join(errored, left=True) 

93 

94 for job in jobs_summary.fetch(as_dict=True): 

95 job_status.append({ 

96 'table': f'{pipeline_module.__name__.split(".")[-1]}.{dj.utils.to_camel_case(job.pop("table_name"))}', 

97 **job}) 

98 

99 if not job_status: 

100 print('No jobs under process (0 reserved jobs)') 

101 return 

102 

103 job_status_df = pd.DataFrame(job_status).set_index('table') 

104 job_status_df.fillna(0, inplace=True) 

105 job_status_df = job_status_df.astype({"reserve_count": int, "error_count": int})
